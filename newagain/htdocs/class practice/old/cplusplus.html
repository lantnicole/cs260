#include iostream
#include math.h
using namespace std;

int main() {
const int PEOPLE_PER_LARGE = 7;
const int PEOPLE_PER_MED = 3;
int numLargePizza = 0;
int numMedPizza = 0;
int numSmallPizza = 0;
int numTotalGuest = 0;
const int LARGE_RADIUS = 10;
const int MED_RADIUS = 8;
const int SMALL_RADIUS = 6;
double largeArea = 0;
double medArea = 0;
double smallArea = 0;
double totalArea = 0;
double individArea = 0.0;
double numTip = 0.0;
double decimalTip = 0.0;
double realTip = 0.0;
const double TIP_CONVERTER = 100;
const double LARGE_PRICE = 14.68;
const double MED_PRICE = 11.48;
const double SMALL_PRICE = 7.28;
double firstTotalPrice = 0;
double totalPrice = 0.0;
int realTotalPrice = 0;
const double PI = 3.14159265;

// COUNT YOUR MANY PIZZAS
cout << "Please enter the number of guests: ";
cin >> numTotalGuest;
cout << endl;

//find large
numLargePizza = numTotalGuest / PEOPLE_PER_LARGE;
cout << numLargePizza << " large pizzas, ";


//find med
numMedPizza = (numTotalGuest % PEOPLE_PER_LARGE) / PEOPLE_PER_MED;
cout << numMedPizza << " medium pizzas, ";

//find small
cout << "and ";
numSmallPizza = numTotalGuest - ((PEOPLE_PER_LARGE * numLargePizza) +
(PEOPLE_PER_MED * numMedPizza));
cout << numSmallPizza << " small pizzas will be needed." << endl;

// PART 2


//find areas individually and totally
largeArea = numLargePizza * (PI * pow(LARGE_RADIUS, 2));
medArea = numMedPizza * (PI * pow(MED_RADIUS, 2));
smallArea = numSmallPizza * (PI * pow(SMALL_RADIUS, 2));
totalArea = largeArea + medArea + smallArea;

//output and find individual
cout << endl << "A total of " << totalArea << " square inches of pizza will be purchased ";
individArea = totalArea / numTotalGuest;
cout << "(" << individArea << " per guest)." << endl;


//Part 3

//find price without tip
firstTotalPrice = (numLargePizza * LARGE_PRICE) + (numMedPizza * MED_PRICE) + (numSmallPizza * SMALL_PRICE);

//find tip
cin >> numTip;
decimalTip = numTip / TIP_CONVERTER;
realTip = firstTotalPrice * decimalTip;

//find total price
totalPrice = realTip + firstTotalPrice;
realTotalPrice = round(totalPrice);

//output
cout << endl << "Please enter the tip as a percentage (i.e. 10 means 10%): " << endl;
cout << "The total cost of the event will be: $" << realTotalPrice << endl;

system("pause");
return 0;
}
